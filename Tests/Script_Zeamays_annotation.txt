#The bowtie2 and hisat2 indexes were build under with this new references.

grep -v '#' Zea_mays.AGPv4.40.gff3 |awk -F "\t" '{$1= "chr"$1;print$0;}' OFS="\t" >Zea_mays.AGPv4.40_mod.gff3
grep -P 'ncRNA_gene.+(biotype=rRNA|biotype=tRNA)' Zea_mays.AGPv4.40_mod.gff3 > Zmay_rtM_RNA.gff3
grep -P '\tchromosome\t' Zea_mays.AGPv4.40_mod.gff3 | grep -Pv 'chr\d+|chrX|chrY' >>Zmay_rtM_RNA.gff3
awk '{print $1,$4,$5,$9,$7}' OFS="\t" Zmay_rtM_RNA.gff3 >Zmay_rtM_RNA.bed
